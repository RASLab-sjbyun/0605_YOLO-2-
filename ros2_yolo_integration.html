01

Workspace & Package 생성

ros 2 workspace · package setup

ROS 2 기반 YOLO 구동 개요
  • YOLO가 ROS 2 기반 시스템에 연동되어 구동되기 위해서는 YOLO 네트워크를 ROS 2 문법에 반영하는 과정이 필요함
  • YOLO는 이미지를 입력으로 받아 객체 및 좌표를 추정하므로 이미지 입력이 필수적임
  • RealSense 카메라로부터 이미지 토픽을 Subscribe하여 YOLO에 전달하는 구조로 구현
전체 노드 아키텍처
📷 RealSense
/camera/.../image_raw
→
🧠 YOLO Node
yolo_ws / yolo_util
→
🖼 Image Topic
/yolo/detected/image
+
📊 Result Topic
/yolo/detected/result

💡
WORKSPACE는 통상적으로 HOME(~)에 생성하며, package는 workspace 안의 src 폴더에 생성되어야 하므로 함께 생성합니다.
bash
mkdir -p ~/yolo_ws/src

📦

import rclpy as rp
from rclpy.node import Node
from sensor_msgs.msg import Image
from cv_bridge import CvBridge
import cv2
from ultralytics import YOLO


class YoloSimpleNode(Node):

    def __init__(self):
        super().__init__('YOLO_SIMPLE')
        self.bridge = CvBridge()
        self.subscription = self.create_subscription(
            Image,
            '/camera/camera/color/image_raw',
            self.callback,
            10
        )
        self.model = YOLO("yolov8n.pt")

    def callback(self, msg):
        cv_image = self.bridge.imgmsg_to_cv2(msg, 'bgr8')
        output = self.model(cv_image)
        annotated_img = output[0].plot()
        cv2.imshow("YOLO Detected", annotated_img)
        cv2.waitKey(1)


def main(args=None):
    rp.init(args=args)
    node = YoloSimpleNode()
    try:
        rp.spin(node)
    except KeyboardInterrupt:
        pass
    node.destroy_node()
    rp.shutdown()


if __name__ == '__main__':
    main()

yolo_image_pub — 결과 이미지 Publish

section 1.4 · annotated image publish · rviz2 확인

이 Node의 역할
  • 1.3의 노드는 결과 이미지를 단순 시각화만 하며 특정 토픽을 Publish하지 않음

import rclpy as rp
from rclpy.node import Node
from sensor_msgs.msg import Image
from cv_bridge import CvBridge
import cv2
from ultralytics import YOLO


class YoloImagePubNode(Node):

    def __init__(self):
        super().__init__('YOLO_IMAGE_PUB')
        self.bridge = CvBridge()
        self.subscription = self.create_subscription(
            Image,
            '/camera/camera/color/image_raw',
            self.callback,
            10
        )
        self.publisher = self.create_publisher(Image, '/yolo/detected/image', 10)
        self.model = YOLO("yolov8n.pt")

    def callback(self, msg):
        cv_image = self.bridge.imgmsg_to_cv2(msg, 'bgr8')
        output = self.model(cv_image)
        annotated_img = output[0].plot()

        pub_msg = self.bridge.cv2_to_imgmsg(annotated_img, encoding='bgr8')
        pub_msg.header = msg.header
        self.publisher.publish(pub_msg)


def main(args=None):
    rp.init(args=args)
    node = YoloImagePubNode()
    try:
        rp.spin(node)
    except KeyboardInterrupt:
        pass
    node.destroy_node()
    rp.shutdown()


if __name__ == '__main__':
    main()

yolo_result_pub — 수치 데이터 Publish

section 1.6 · YoloDetectionArray publish · topic echo

이 Node의 역할
  • YOLO 결과를 이미지가 아닌 정량적 수치 데이터로 Publish함
  • 각 탐지 객체의 bounding box 좌표, class ID, confidence를 YoloDetectionArray 타입으로 발행
1.6.1 Node 파일 생성 및 작성
bash · 파일 생성
cd ~/yolo_ws/src/yolo_util/yolo_util/
gedit yolo_result_pub.py
python · yolo_result_pub.py · 전체 코드
import rclpy as rp
from rclpy.node import Node
from sensor_msgs.msg import Image
from cv_bridge import CvBridge
import cv2
from ultralytics import YOLO
from yolo_msgs.msg import YoloDetection
from yolo_msgs.msg import YoloDetectionArray


class YoloResultPubNode(Node):

    def __init__(self):
        super().__init__('YOLO_RESULT_PUB')
        self.bridge = CvBridge()
        self.subscription = self.create_subscription(
            Image,
            '/camera/camera/color/image_raw',
            self.callback,
            1
        )
        self.publisher = self.create_publisher(YoloDetectionArray, '/yolo/detected/result', 10)
        self.model = YOLO("yolov8n.pt")

    def callback(self, msg):
        cv_image = self.bridge.imgmsg_to_cv2(msg, 'bgr8')
        output = self.model(cv_image)
        annotated_img = output[0].plot()
        result = output[0]

        detection_array = YoloDetectionArray()
        detection_array.detections = []
        detection_array.header = msg.header

        for box in result.boxes:
            x1, y1, x2, y2 = box.xyxy[0]
            confidence = float(box.conf[0])
            class_id = int(box.cls[0])
            class_name = self.model.names[class_id]

            det_msg = YoloDetection()
            det_msg.x1 = float(x1)
            det_msg.y1 = float(y1)
            det_msg.x2 = float(x2)
            det_msg.y2 = float(y2)
            det_msg.confidence = confidence
            det_msg.class_id = class_id
            det_msg.class_name = str(class_name)
            detection_array.detections.append(det_msg)

        self.publisher.publish(detection_array)
        cv2.imshow("YOLO Detected", annotated_img)
        cv2.waitKey(1)


def main(args=None):
    rp.init(args=args)
    node = YoloResultPubNode()
    try:
        rp.spin(node)
    except KeyboardInterrupt:
        pass
    node.destroy_node()
    rp.shutdown()


if __name__ == '__main__':
    main()

            det_msg.y1 = float(y1)
            det_msg.x2 = float(x2)
            det_msg.y2 = float(y2)
            det_msg.confidence = confidence
            det_msg.class_id = class_id
            det_msg.class_name = str(class_name)
            detection_array.detections.append(det_msg)

        self.result_publisher.publish(detection_array)
        self.image_publisher.publish(pub_img_msg)


def main(args=None):
    rp.init(args=args)
    node = YoloAllPubNode()
    try:
        rp.spin(node)
    except KeyboardInterrupt:
        pass
    node.destroy_node()
    rp.shutdown()


if __name__ == '__main__':
    main()
